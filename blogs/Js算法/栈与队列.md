
### 栈和队列
---
* 使用栈实现队列

* 判断括号是否是匹配（“(){()[{}]} ()({}) {()}[]{()} [{{[()]}}]”）
```js
const map = {
  '(': ')',
  '[': ']',
  '{': '}'
}

const isValid = (str) => {
  const stack = [];

  for (let i = 0; i < str.length; i++) {
    const top = stack.length > 0 ? stack[stack.length - 1] : null
    if (str[i] === ')' || str[i] === ']' || str[i] === '}') {
      if (map[top] !== str[i]) {
        return false;
      } else {
        stack.pop();
      }
    } else if (map[str[i]]) {
      stack.push(str[i]);
    }
  }
  return stack.length === 0;
}

const str = '(){()[{}]} ()({}) {()}[]{()} [{{[()]}}]';
console.log(isValid(str))
```
