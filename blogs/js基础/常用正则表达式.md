# 常见正则汇总

```js
// 匹配手机号:现在虚拟号码的使用，所以第二位也不似以前的3|4|5|8的固定，所以此处是选择的3-9
let telReg = /^1[3-9][0-9]{9}$/
// 匹配身份证
let IdCardReg = /^[1-9]\d{5}(18|19|20)\d{2}((0[1-9])|(1[0-2]))(([0-2][1-9])|10|20|30|31)\d{3}[0-9Xx]$/
// 匹配邮箱
let mailReg =  /^[a-zA-Z0-9_-]+@[a-zA-Z0-9_-]+(.[a-zA-Z0-9_-]+)+$/

// 匹配颜色
let colorReg = /^#([A-Fa-f0-9]{6}|[A-Fa-f0-9]{3})/g

// 匹配ip地址
// 匹配html标签
// 匹配中文
let chineseReg = /[\u4e00-\u9FFF]/g
```
